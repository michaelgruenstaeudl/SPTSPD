## Generating package
python2 setup.py sdist bdist_wheel

## Uploading package to PyPI test server and installing it from there
twine upload --repository-url https://test.pypi.org/legacy/ dist/*
python2 -m pip install --index-url https://test.pypi.org/simple/ EMBL2checklists

## Uploading package to PyPI main server
twine upload dist/*
pip2 install EMBL2checklists
